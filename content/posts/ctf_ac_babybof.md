We have a `win()` function that prints the flag:

```c
void win(void)
{
  char *bytesRead;
  char buffer [136];
  FILE *fh;
  
  fh = fopen("flag.txt","r");
  if (fh == (FILE *)0x0) {
    puts("Flag missing.");
    fflush(stdout);
    exit(1);
  }
  bytesRead = fgets(buffer,0x80,fh);
  if (bytesRead != (char *)0x0) {
    puts(buffer);
    fflush(stdout);
  }
  fclose(fh);
  exit(0);
}
```
